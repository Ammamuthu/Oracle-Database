1.)  Create a Configuration on Primary Database 

CREATE CONFIGURATION 'my_config' AS PRIMARY DATABASE IS 'CDB_P' CONNECT IDENTIFIER IS 'CDB_P';

2.) Add a Standby DATABASE 

ADD DATABASE 'CDB_D' AS CONNECT IDENTIFIER IS 'CDB_D' MAINTAINED AS PHYSICAL;

3.)  Start the Configuration 

ENABLE CONFIGURATION;

****      Run it in a DGMGRL     ****

EDIT DATABASE 'CDB_P' set state='TRANSPORT-ON';
EDIT DATABASE 'CDB_D' set state='APPLY-ON';
EDIT DATABASE 'CDB_P' set PROPERTY 'TransportDisconnectedThreshold'=0;
EDIT DATABASE 'CDB_D' SET PROPERTY 'TransportDisconnectedThreshold'=0;
EDIT DATABASE 'CDB_P'SET PROPERTY 'LogXptMode'='SYNC';
EDIT DATABASE 'CDB_D' SET PROPERTY 'LogXptMode'='SYNC';
EDIT DATABASE 'CDB_P'SET PROPERTY 'REDOCOMPRESSION'='ENABLE';
EDIT DATABASE 'CDB_D' SET PROPERTY 'REDOCOMPRESSION'='ENABLE';
EDIT CONFIGURATION SET PROTECTION MODE AS MAXPERFORMANCE;
EDIT DATABASE 'CDB_P' SET PROPERTY 'ArchiveLagTarget'=1200;
EDIT DATABASE 'CDB_P' SET PROPERTY 'LogArchiveMaxProcesses'=10;
EDIT DATABASE 'CDB_P' SET PROPERTY 'LogArchiveMinSucceedDest'=1;
EDIT DATABASE 'CDB_D' SET PROPERTY 'ArchiveLagTarget'=1200;
EDIT DATABASE 'CDB_D' SET PROPERTY 'LogArchiveMaxProcesses'=10;
EDIT DATABASE 'CDB_D' SET PROPERTY 'LogArchiveMinSucceedDest'=1;
